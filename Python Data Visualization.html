df2 = df1.loc[(df1['Country'] == 'United States')].copy()
plt.title('Percentage of orders in each Product_Category in the United States')
explode_list = [0,0,0.1]
df2['Product_Category'].value_counts().plot(kind='pie', 
                                            figsize = (5,5),
                                            autopct='%0.1f%%',
                                            startangle=90,
                                            shadow=True,
                                            explode = explode_list,
                                            colors = sns.color_palette('Set3'))
plt.legend(df2['Product_Category'].value_counts().index, loc = 'upper right', bbox_to_anchor=(1.5,1))
plt.show()
